# helper functions ----
render_widget <- function(w, f){
  p <- glue("figures/{f}.png")
  if (!file.exists(p)){
    h <- glue("figures/{f}.html")
    htmlwidgets::saveWidget(w, h)
    webshot2::webshot(h, p)
  }
  if (knitr::is_html_output()){
    return(w)
  } else {
    return(knitr::include_graphics(p))
  }
}
